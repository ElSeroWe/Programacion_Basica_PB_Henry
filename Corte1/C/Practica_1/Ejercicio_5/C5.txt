#include <stdio.h>
#include <stdlib.h>

int main(){
	int a, b, c, max, min;
	printf("Digite 3 numeros\n");
	scanf("%d, %d, %d", &a, &b, &c);
	max=a;
	min=a;
	if (b>max){
		max=b;
	} else {
		max=max; }
	if (c>max){
		max=c;
	} else {
		max=max; }	
	if (b<min) {
		min=b;
	} else {
		min=min; }
	if (c<min){
		min=c;
	} else {
		min=min; }
	printf("El valor maximo es: %d\nEl valor minimo es: %d", max, min);	
}
