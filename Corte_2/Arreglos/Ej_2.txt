int main() {	
	int n;
	int suma=0;
	int M1[n][n];
	int i,j;
	for (i=0;i<n;i++){
		for(j=0;j<n;j++){
			printf("Digite valores para la matriz:\n");
			scanf("%d", &M1[i][j]);
		}
	}
	for(i=0;i<n;i++){
		for(j=0;j<n;j++){
			printf(" %d", M1[i][j]);
		}
		printf("\n");
	}
	for(i=0;i<n;i++){
		for(j=0;j<n;j++){
			suma=suma+M1[i][j];
		}
	}
	printf("La suma es: %d", suma);
	return 0;
}
