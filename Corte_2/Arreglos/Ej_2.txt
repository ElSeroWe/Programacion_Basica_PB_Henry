#include <stdio.h>
#include <stdlib.h>
int main(){
	int i,xd;
	printf("Digite la cantidad de elementos que va a tener el vector: ");
	scanf("%d",&xd);
	int V1[xd];
	for(i=0;i<xd;i++){
		printf("Digite el valor del arreglo: \n");
		scanf("%d",&V1[i]);
		if (V1[i]%2==0){
			printf("Numero aceptado\n");
		} else{
			printf("No se aceptan numeros impares\n");
			i--;
		}}
	printf("Los valores del arreglo son: ");
	for(i=0;i<xd;i++){
		printf("%d", V1[i]);
	}
	return 0;
}
