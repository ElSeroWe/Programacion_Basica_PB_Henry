int main() {
	int suma=0;
	int M1[3][3];
	int i,j;
	for (i=0;i<3;i++){
		for(j=0;j<3;j++){
			printf("Digite valores para la matriz:\n");
			scanf("%d", &M1[i][j]);
		}
	}
	for(i=0;i<3;i++){
		for(j=0;j<3;j++){
			printf(" %d", M1[i][j]);
		}
		printf("\n");
	}
	for(i=0;i<3;i++){
		for(j=0;j<3;j++){
			suma=suma+M1[i][j];
		}
	}
	printf("La suma es: %d", suma);
	return 0;
}
